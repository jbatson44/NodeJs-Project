CREATE TABLE users (
	userId serial PRIMARY KEY,
	username varchar(80), 
	password varchar(80), 
	firstName varchar(80),
	lastName varchar(80),
	email varchar(80),
	state varchar(80), 
	city varchar(80), 
	gender varchar(80)
);
